## Evento `messages.received`

Disparado quando uma mensagem é recebida.

### Schema Base

```typescript
interface WebhookPayload {
  event: "messages.received";
  sessionId: string;
  timestamp: string; // ISO 8601
  traceId: string;
  data: MessageData;
}

interface MessageData {
  id: string;
  type: MessageType;
  message: string;
  timestamp: number; // Unix timestamp (ms)
  fromMe: boolean;
  is_group: boolean;
  group_name: string;
  from: Contact;
  to: Contact;
  from_name: string;
  media_url: string | null;
  media_data?: MediaData;
  data?: ContactData | LocationData | ReactionData | PollUpdateData;
  context_info?: ContextInfo;
  external_ad_reply?: ExternalAdReply;
}

type MessageType =
  | "text"
  | "image"
  | "video"
  | "audio"
  | "document"
  | "sticker"
  | "contact"
  | "location"
  | "reaction"
  | "poll_update";
```

### Types

```typescript
interface Contact {
  jid: string;      // 5511999999999@s.whatsapp.net
  lid: string;      // ID interno
  name: string;
}

interface MediaData {
  url: string;
  file_size: number;
  mime_type: string;
  // Image/Video
  width?: number;
  height?: number;
  caption?: string;
  // Audio
  duration?: number;
  is_ptt?: boolean;  // Push-to-talk (voice message)
  // Document
  filename?: string;
  // Sticker
  is_animated?: boolean;
}

interface ContactData {
  contact_name: string;
  contact_phone: string;
  display_name: string;
  vcard: string;
}

interface LocationData {
  degrees_latitude: number;
  degrees_longitude: number;
  name: string;
  address: string;
  thumbnail: string; // Base64
}

interface ReactionData {
  original_message_id: string;
  reaction_text: string;      // Emoji
  reaction_timestamp: number;
  target_participant: string;
}

interface PollUpdateData {
  poll_creation_message_key: {
    remoteJID: string;
    fromMe: boolean;
    ID: string;
  };
  vote: {
    encPayload: string;
    encIV: string;
  };
}

interface ContextInfo {
  // Resposta a mensagem
  stanza_id?: string;
  quoted_message_id?: string;
  quoted_message?: {
    body: string;
    type: string;
  };
  participant?: string;
  // CTWA (Click to WhatsApp Ads)
  entry_point_conversion_app?: string;
  entry_point_conversion_source?: string;
  external_ad_reply?: ExternalAdReply;
}

interface ExternalAdReply {
  body: string;
  title: string;
  ctwa_clid: string;
  source_id: string;
  source_type: string;
  source_url: string;
  media_type: number;
  media_url: string;
  thumbnail_url: string;
  render_larger_thumbnail: boolean;
  show_ad_attribution: boolean;
}
```

## Identificando Tipos

| Condição | Tipo |
|----------|------|
| `data.type === "text"` | Mensagem de texto |
| `data.type === "image"` | Imagem |
| `data.type === "video"` | Vídeo |
| `data.type === "audio"` | Áudio/PTT |
| `data.type === "document"` | Documento |
| `data.type === "sticker"` | Sticker |
| `data.type === "contact"` | Contato |
| `data.type === "location"` | Localização |
| `data.type === "reaction"` | Reação |
| `data.type === "poll_update"` | Atualização de enquete |
| `data.context_info?.quoted_message` | É resposta |
| `data.context_info?.entry_point_conversion_source === "ctwa_ad"` | Veio de anúncio |

## Media URLs

Arquivos de mídia são automaticamente salvos no S3 e disponibilizados em `data.media_url`.

| Tipo | Extensão | Exemplo |
|------|----------|---------|
| image | .jpg, .png | `https://s3.../image/{id}.jpg` |
| video | .mp4 | `https://s3.../video/{id}.mp4` |
| audio | .ogg | `https://s3.../audio/{id}.ogg` |
| document | .pdf, .docx | `https://s3.../application/{id}.pdf` |
| sticker | .webp | `https://s3.../image/{id}.webp` |
