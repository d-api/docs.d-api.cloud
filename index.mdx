## Evento `messages.received`

Disparado quando uma mensagem é recebida.

### Payload

<ResponseField name="event" type="string" required>
  Sempre `"messages.received"`
</ResponseField>

<ResponseField name="sessionId" type="string" required>
  ID da sessão que recebeu a mensagem
</ResponseField>

<ResponseField name="timestamp" type="string" required>
  Timestamp ISO 8601 do recebimento
</ResponseField>

<ResponseField name="traceId" type="string" required>
  ID único para rastreamento
</ResponseField>

<ResponseField name="data" type="object" required>
  <Expandable title="Campos de data">
    <ResponseField name="id" type="string" required>
      ID único da mensagem
    </ResponseField>
    <ResponseField name="type" type="string" required>
      `text` | `image` | `video` | `audio` | `document` | `sticker` | `contact` | `location` | `reaction` | `poll_update`
    </ResponseField>
    <ResponseField name="message" type="string" required>
      Conteúdo textual da mensagem
    </ResponseField>
    <ResponseField name="timestamp" type="number" required>
      Unix timestamp em milissegundos
    </ResponseField>
    <ResponseField name="fromMe" type="boolean" required>
      Se a mensagem foi enviada por você
    </ResponseField>
    <ResponseField name="is_group" type="boolean" required>
      Se é mensagem de grupo
    </ResponseField>
    <ResponseField name="group_name" type="string" required>
      Nome do grupo (vazio se não for grupo)
    </ResponseField>
    <ResponseField name="from_name" type="string" required>
      Nome do remetente
    </ResponseField>
    <ResponseField name="media_url" type="string | null" required>
      URL do arquivo no S3 (para mensagens com mídia)
    </ResponseField>
    <ResponseField name="from" type="object" required>
      <Expandable title="Campos de from">
        <ResponseField name="jid" type="string" required>
          WhatsApp ID (ex: `5511999999999@s.whatsapp.net`)
        </ResponseField>
        <ResponseField name="lid" type="string" required>
          ID interno do WhatsApp
        </ResponseField>
        <ResponseField name="name" type="string" required>
          Nome do contato
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="to" type="object" required>
      <Expandable title="Campos de to">
        <ResponseField name="jid" type="string" required>
          WhatsApp ID do destinatário
        </ResponseField>
        <ResponseField name="name" type="string" required>
          Nome do destinatário
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="media_data" type="object">
      Presente em mensagens de mídia (image, video, audio, document, sticker)
      <Expandable title="Campos de media_data">
        <ResponseField name="url" type="string" required>
          URL original do WhatsApp (temporária)
        </ResponseField>
        <ResponseField name="file_size" type="number" required>
          Tamanho em bytes
        </ResponseField>
        <ResponseField name="mime_type" type="string" required>
          MIME type do arquivo
        </ResponseField>
        <ResponseField name="width" type="number">
          Largura (image/video/sticker)
        </ResponseField>
        <ResponseField name="height" type="number">
          Altura (image/video/sticker)
        </ResponseField>
        <ResponseField name="caption" type="string">
          Legenda (image/video/document)
        </ResponseField>
        <ResponseField name="duration" type="number">
          Duração em segundos (audio/video)
        </ResponseField>
        <ResponseField name="is_ptt" type="boolean">
          Push-to-talk / mensagem de voz (audio)
        </ResponseField>
        <ResponseField name="filename" type="string">
          Nome do arquivo (document)
        </ResponseField>
        <ResponseField name="is_animated" type="boolean">
          Se é sticker animado (sticker)
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="data" type="object">
      Dados específicos para contact, location, reaction e poll_update
      <Expandable title="ContactData (type=contact)">
        <ResponseField name="contact_name" type="string" required>
          Nome do contato
        </ResponseField>
        <ResponseField name="contact_phone" type="string" required>
          Telefone do contato
        </ResponseField>
        <ResponseField name="display_name" type="string" required>
          Nome de exibição
        </ResponseField>
        <ResponseField name="vcard" type="string" required>
          VCard completo
        </ResponseField>
      </Expandable>
      <Expandable title="LocationData (type=location)">
        <ResponseField name="degrees_latitude" type="number" required>
          Latitude
        </ResponseField>
        <ResponseField name="degrees_longitude" type="number" required>
          Longitude
        </ResponseField>
        <ResponseField name="name" type="string" required>
          Nome do local
        </ResponseField>
        <ResponseField name="address" type="string" required>
          Endereço
        </ResponseField>
        <ResponseField name="thumbnail" type="string" required>
          Imagem em Base64
        </ResponseField>
      </Expandable>
      <Expandable title="ReactionData (type=reaction)">
        <ResponseField name="original_message_id" type="string" required>
          ID da mensagem original
        </ResponseField>
        <ResponseField name="reaction_text" type="string" required>
          Emoji da reação
        </ResponseField>
        <ResponseField name="reaction_timestamp" type="number" required>
          Timestamp da reação
        </ResponseField>
        <ResponseField name="target_participant" type="string" required>
          Participante alvo (em grupos)
        </ResponseField>
      </Expandable>
      <Expandable title="PollUpdateData (type=poll_update)">
        <ResponseField name="poll_creation_message_key" type="object" required>
          Referência à enquete original
        </ResponseField>
        <ResponseField name="vote" type="object" required>
          Dados criptografados do voto
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="context_info" type="object">
      Presente em respostas e mensagens CTWA
      <Expandable title="Campos de context_info">
        <ResponseField name="stanza_id" type="string">
          ID da mensagem citada
        </ResponseField>
        <ResponseField name="quoted_message_id" type="string">
          ID da mensagem citada
        </ResponseField>
        <ResponseField name="quoted_message" type="object">
          <Expandable title="Campos">
            <ResponseField name="body" type="string">Conteúdo da mensagem citada</ResponseField>
            <ResponseField name="type" type="string">Tipo da mensagem citada</ResponseField>
          </Expandable>
        </ResponseField>
        <ResponseField name="participant" type="string">
          Autor da mensagem citada
        </ResponseField>
        <ResponseField name="entry_point_conversion_app" type="string">
          App de origem CTWA (ex: `facebook`)
        </ResponseField>
        <ResponseField name="entry_point_conversion_source" type="string">
          Fonte CTWA (ex: `ctwa_ad`)
        </ResponseField>
        <ResponseField name="external_ad_reply" type="object">
          Dados do anúncio (ver ExternalAdReply)
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="external_ad_reply" type="object">
      Dados do anúncio CTWA (duplicado no root de data)
      <Expandable title="Campos de external_ad_reply">
        <ResponseField name="body" type="string" required>
          Texto do anúncio
        </ResponseField>
        <ResponseField name="title" type="string" required>
          Título do anúncio
        </ResponseField>
        <ResponseField name="ctwa_clid" type="string" required>
          Click ID para atribuição
        </ResponseField>
        <ResponseField name="source_id" type="string" required>
          ID do anúncio
        </ResponseField>
        <ResponseField name="source_type" type="string" required>
          Tipo (ex: `ad`)
        </ResponseField>
        <ResponseField name="source_url" type="string" required>
          URL do anúncio
        </ResponseField>
        <ResponseField name="media_type" type="number" required>
          Tipo de mídia
        </ResponseField>
        <ResponseField name="media_url" type="string" required>
          URL da mídia do anúncio
        </ResponseField>
        <ResponseField name="thumbnail_url" type="string" required>
          URL da thumbnail
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## Identificando Tipos

| Condição | Tipo |
|----------|------|
| `data.type === "text"` | Mensagem de texto |
| `data.type === "image"` | Imagem |
| `data.type === "video"` | Vídeo |
| `data.type === "audio"` | Áudio/PTT |
| `data.type === "document"` | Documento |

## Media URLs

Arquivos de mídia são automaticamente salvos no S3 e disponibilizados em `data.media_url`.

| Tipo | Extensão | Exemplo |
|------|----------|---------|
| image | .jpg, .png | `https://s3.../image/{id}.jpg` |
| video | .mp4 | `https://s3.../video/{id}.mp4` |
| audio | .ogg | `https://s3.../audio/{id}.ogg` |
| document | .pdf, .docx | `https://s3.../application/{id}.pdf` |
| sticker | .webp | `https://s3.../image/{id}.webp` |
